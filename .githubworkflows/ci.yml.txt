name: Python CI Pipeline

on:
  push:
    branches:
      - kernelridge # Pipeline triggers only when code is pushed to this branch

jobs:
  build-and-test:
    runs-on: ubuntu-latest # Standard runner for CI/CD

    steps:
    - name: Checkout Repository
      uses: actions/checkout@v4

    - name: Set up Python 3.9
      uses: actions/setup-python@v4
      with:
        python-version: '3.9' # Use the same Python version as your environment

    - name: Install Dependencies
      run: |
        python -m pip install --upgrade pip
        pip install -r requirements.txt
    
    - name: Run Decision Tree Model (train.py)
      run: |
        echo "Running first model (Decision Tree)..."
        python train.py
        
    - name: Run Kernel Ridge Model (train2.py)
      run: |
        echo "Running second model (Kernel Ridge)..."
        python train2.py
